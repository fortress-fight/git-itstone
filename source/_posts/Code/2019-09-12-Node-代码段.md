## Node-代码段

### create_file & create_dir

```js
const create_dir = path => {
    try {
        fs.statSync(path, (err, state) => {
            if (!state.isDirectory()) {
                fs.mkdirSync(path);
            }
        });
    } catch (err) {
        fs.mkdirSync(path);
    }
};

const create_file = path => {
    try {
        fs.statSync(path, (err, state) => {
            if (!state.isFile()) {
                fs.writeFileSync(path, "");
            }
        });
    } catch (err) {
        fs.writeFileSync(path, "");
    }
};
```
